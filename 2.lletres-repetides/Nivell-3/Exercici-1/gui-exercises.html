Exercici 2

  • Al bucle, si la lletra és una vocal imprimeix a la consola: ‘He trobat la VOCAL: __’. Sinó, imprimeix: ‘He trobat la CONSONTANT: __’.
  • Si trobes un número, mostra per pantalla: ‘Els noms de persones no contenen el número: __’
Exercici 2 Solution 
        
          function isVowel(letter) {
            return "aeiouAEIOU".indexOf(letter) != -1;
          }

          function isConsonant(letter) {
            return "bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ".indexOf(letter) != -1;
          }

          function isNumber(numb) {
            return "0123456789".indexOf(numb) != -1;
          }

          let nom = ["d", "4", "N", "I", "E", "L", "-"];

          for (let i = 0; i < nom.length; i++) {
            if (isVowel(nom[i])) {
              document.write("Vowel: " + nom[i] + " found");
            } else if (isConsonant(nom[i])) {
              document.write("Consonant: " + nom[i] + " found");
            } else if (isNumber(nom[i])) {
              document.write("Number: " + nom[i] + " cannot be part of a name");
            } else {
              document.write("Character: " + nom[i] + " cannot be part of a name");
            }
          }
Output

Exercici 3

Emmagatzemar en un Map les lletres de l'array y el nombre de vegades que apareixen.
Exercici 3 Solution 
        
          const name1 = "DANIELITONCIO";

        	let name_map = new Map();

        	for (let c of name1) {
        		if (!name_map.has(c)) {
        			name_map.set(c, 1);
        		} else {
        			name_map.set(c, name_map.get(c) + 1);
        		}
        	}
        	document.write("Name: ", name1);
Output
